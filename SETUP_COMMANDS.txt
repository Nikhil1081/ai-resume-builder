# SETUP COMMANDS FOR YOUR PROJECT

## Step 1: Create .env file with your OpenAI API key
echo OPENAI_API_KEY=your_actual_openai_key_here > .env
echo PORT=5000 >> .env

## Step 2: Run the setup script
setup.bat

## Step 3: Start the application
python app.py

## Your app will be available at http://localhost:5000

## ============================================
## DEPLOY TO GITHUB AND RAILWAY
## ============================================

## Step 4: Initialize Git Repository
git init
git add .
git commit -m "Initial commit: AI Resume & Portfolio Builder"

## Step 5: Create GitHub repository
## Go to https://github.com/new and create a repository named "ai-resume-builder"

## Step 6: Push to GitHub
git remote add origin https://github.com/Nikhil1081/ai-resume-builder.git
git branch -M main
git push -u origin main

## Step 7: Deploy on Railway
## 1. Go to https://railway.app
## 2. Sign in with GitHub
## 3. Click "New Project"
## 4. Select "Deploy from GitHub repo"
## 5. Choose "Nikhil1081/ai-resume-builder"
## 6. Add environment variable:
##    Key: OPENAI_API_KEY
##    Value: your_openai_api_key
## 7. Railway will automatically deploy!

## Your live app URL will be: https://your-app-name.up.railway.app
